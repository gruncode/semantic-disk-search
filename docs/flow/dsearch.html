1#!/bin/bash
2# ================================================================================
3# dsearch — Full-disk semantic search CLI
4# ================================================================================
5#
6# PURPOSE
7# Entry point for semantic search across ~/DATA. Routes to one of 3 modes:
8# - Terminal mode (default) — colored text output, quick ranked list
9# - HTML mode (-w) — sortable/filterable HTML table in browser
10# - Answer mode (-CLexpand-...) — full 6-step pipeline → cited answer
11#
12# INPUTS
13# Command-line flags (see usage() below) + positional query words
14#
15# OUTPUTS
16# Terminal mode: colored rows to stdout (score, path, snippet)
17# HTML mode: /tmp/dsearch_results.html, opened via xdg-open
18# Answer mode: delegates to dsearch-answer (which prints ANSWER + SOURCES)
19#
20# RESOURCES
21# ~/.config/dsearch/.env — COHERE_API_KEY (loaded at startup)
22# $CHROMADB_DIR — ChromaDB persistent collection 'fulldisk'
23# $VENV_GTE/ — Python venv (has cohere + chromadb libs)
24#
25# RELATED FILES
26# dsearch-answer — Full CLexpand→CLrank→CLanswer pipeline (6 steps)
27# extract_chunks.py — Builds ChromaDB from disk files (one-time indexing)
28# docs/index.html — Diagram, clickable links back to dsearch-answer
29# ================================================================================

40Usage: dsearch [-n NUM] [-w] [-CLexpand-CLrank-CLanswer] query words

48 rank files → final Claude reads top 15 originals → answer (~3-4 min)

63# ── Arg parsing ──────────────────────────────────────────────
64NRESULTS=50 # Default: 50 results for quick browsing
65WEBMODE=0 # -w flag → HTML output
66ANSWERMODE=0 # -CLexpand-CLrank-CLanswer → delegate to dsearch-answer
67DEEPMODE="" # -deep (only meaningful with ANSWERMODE)
68while [ $# -gt 0 ]; do
69 case "$1" in
70 -h|--help) usage ;;
71 -n) NRESULTS="$2"; shift 2 ;;
72 -w) WEBMODE=1; shift ;;
73 # Answer mode forces NRESULTS=100 (pipeline needs 100 files for 10×10 agents)
74 -CLexpand-CLrank-CLanswer) ANSWERMODE=1; NRESULTS=100; shift ;;
75 -deep) DEEPMODE="-deep"; shift ;;
76 *) break ;; # First non-flag arg → query starts here
77 esac
78done

101# Single-query Cohere embed (async version in dsearch-answer handles batch)
102co = cohere.ClientV2(api_key=os.environ['COHERE_API_KEY'])
103resp = co.embed(texts=[q], model='embed-multilingual-v3.0',
104 input_type='search_query', embedding_types=['float'])

106# ChromaDB: 941K chunks indexed from ~/DATA (built by extract_chunks.py)
107col = chromadb.PersistentClient(
108 path='$CHROMADB_DIR').get_collection('fulldisk')
109results = col.query(query_embeddings=resp.embeddings.float_, n_results=n)

111# Build a word-level highlight regex.
112# Filter 3+ chars: avoids highlighting "a", "of", "το" etc. (too noisy)
113# \b word boundaries work with Greek since Python's re module is Unicode-aware
114words = [w for w in q.split() if len(w) > 2]
115pat = re.compile(r'\b(' + '|'.join(re.escape(w) for w in words) + r')\b',
116 re.IGNORECASE) if words else None

118def hl(text):
119 """HTML-escape then wrap query words in <span class="hl"> for yellow highlight."""
120 text = h.escape(text)
121 if pat:
122 text = pat.sub(r'<span class="hl">\1</span>', text)
123 return text

125def score_cls(s):
126 """Map score 0-1 to CSS class for color coding.
127 sh=green high >=0.6, sm=yellow mid >=0.5, sl=red low <0.5"""
128 if s >= 0.6: return 'sh'
129 if s >= 0.5: return 'sm'
130 return 'sl'

132def fmt_date(ts):
133 try:
134 return time.strftime('%Y-%m-%d', time.localtime(int(ts)))
135 except:
136 return '—'

138# Build one <tr> per result. data-val attributes hold sortable values
139# (e.g. raw score/timestamp) so JS sort doesn't have to re-parse the display text.
140rows = []
141for i, (doc, meta, dist) in enumerate(zip(
142 results['documents'][0], results['metadatas'][0],
143 results['distances'][0])):
144 score = 1 - dist # Cosine distance → similarity
145 path = meta['path']
146 dirn = os.path.dirname(path) # Parent folder (full path)
147 fname = os.path.basename(path)
148 ext = meta.get('ext', '') # File extension (without dot)
149 cat = meta.get('folder', '') # Top-level category (DATA/Finance, etc)
150 mdate = fmt_date(meta.get('mtime', '0')) # Human-readable modification date
151 mts = meta.get('mtime', '0') # Raw epoch seconds (for sorting)
152 cidx = meta.get('chunk_idx', 0) # Which chunk of the file matched
153 ctot = meta.get('total_chunks', 1) # Total chunks in that file
154 snippet = doc[:250] # First 250 chars of matching chunk
155 rows.append(f'''<tr>
156 <td class="c" data-val="{i+1}">{i+1}</td>
157 <td class="c {score_cls(score)}" data-val="{score:.4f}">{score:.3f}</td>
158 <td class="cat">{h.escape(cat)}</td>
159 <td class="folder" data-val="{h.escape(dirn)}"><a href="file://{h.escape(dirn)}">{hl(dirn)}</a></td>
160 <td class="fname"><a href="file://{h.escape(path)}">{hl(fname)}</a></td>
161 <td class="ext">{h.escape(ext)}</td>
162 <td class="c" data-val="{mts}">{mdate}</td>
163 <td class="c chunk">{cidx}/{ctot}</td>
164 <td class="snip" title="{h.escape(doc[:300])}">{hl(snippet)}</td>
165</tr>''')

343# ANSI escape codes for terminal colors
